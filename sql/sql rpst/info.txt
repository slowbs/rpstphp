SELECT i.info_id as id, address, villagename, TAMBONNAME, amphur_name   FROM `info` i 
inner JOIN amphur a on a.amphurcode = i.amphur
inner JOIN tambon t on t.TAMBON = i.amphur
inner JOIN village on village.villagecode = i.village
where i.info_id = 2 group by i.info_id

SELECT i.info_id as id, address, villagename, TAMBONNAME, amphur_name, academic, nurse, dentist,
other, total, updated_time, updated_by
FROM `info` i 
inner JOIN amphur a on a.amphurcode = i.amphur
inner JOIN tambon t on t.TAMBON = i.tambon and t.AMPUR = i.amphur
inner JOIN village on village.villagecode = i.village and village.ampurcode = i.amphur and village.tamboncode = i.tambon
where i.info_id = 2 group by i.info_id
